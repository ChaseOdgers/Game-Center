    function shuffleDeck(deck)
{
  let temp;
  let randomPosition;
  let position = deck.length;
  while (position != 0)
  {
    randomPosition = Math.floor(Math.random() * position);
    position--;
    temp = deck[position];
    deck[position] = deck[randomPosition];
    deck[randomPosition] = temp;
  }
  return deck;
}
