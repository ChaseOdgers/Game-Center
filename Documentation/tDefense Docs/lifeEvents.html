  this.modType = "lifeEvent";
  this.student = student;
  this.length = 30;

  if(getRandomInt(2)==1)
  {
    this.type = "negative";
    this.name = generateRandomBadLifeEventName();
  }
  else

  }

}

  let arr = [
    "Car accident",

    "You died"
  ];
  let i = getRandomInt(arr.length);

}
